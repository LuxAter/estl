377  template <typename T>
378  inline std::string data_fmt(int data[8], std::size_t n,
379  std::string_view type, T args) {
380  std::string fmt = "%";
381  if (type[0] == '-') {
382  fmt = '-' + fmt;
383  type = type.substr(1);
384  }
385  if (data[3] == 0) fmt += '+';
386  if (data[3] == 1) fmt += '-';
387  if (data[3] == 2) fmt += ' ';
388  if (data[1] == int('0')) fmt += '0';
389  if (data[4] != -1 && (data[1] == int('0') || data[1] == -1) &&
390  data[2] == -1)
391  fmt += std::to_string(data[4]);
392  if (data[5] != -1) fmt += '.' + std::to_string(data[5]);
393  fmt += std::string(type);
394  if (fmt.back() == 'b') {
395  fmt.back() = 's';
396  fmt = to_binary<std::string>(n, fmt.c_str(), args);
397  } else {
398  fmt = to_xstring<std::string>(n, fmt.c_str(), args);
399  }
400  if (data[4] != -1 && static_cast<std::size_t>(data[4]) > fmt.size()) {
401  if (data[1] == -1) {
402  data[1] = int(' ');
403  }
404  if (data[2] == 0) {
405  fmt += std::string(data[4] - fmt.size(), data[1]);
406  } else if (data[2] == 1) {
407  fmt = std::string(data[4] - fmt.size(), data[1]) + fmt;
408  } else if (data[2] == 2) {
409  if (fmt[0] == '-' || fmt[0] == '+') {
410  fmt = fmt[0] + std::string(data[4] - fmt.size(), data[1]) +
411  fmt.substr(1);
412  } else {
413  fmt = std::string(data[4] - fmt.size(), data[1]) + fmt;
414  }
415  } else if (data[2] == 3) {
416  fmt = std::string(std::floor((data[4] - fmt.size()) / 2.0), data[1]) +
417  fmt +
418  std::string(std::ceil((data[4] - fmt.size()) / 2.0), data[1]);
419  }
420  }
421  return fmt;
422  }
