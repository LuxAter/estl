1407  _Tp determinant(const estl::matrix::Matrix<_Tp, _Nr, _Nc>& lhs) {
1408  estl::matrix::Matrix<_Tp, _Nr, _Nc> mat(lhs);
1409  _Tp sign = _Tp(1);
1410  for (typename estl::matrix::Matrix<_Tp, _Nr, _Nc>::size_type it = 0;
1411  it != mat.rows() && it != mat.columns(); it++) {
1412  typename estl::matrix::Matrix<_Tp, _Nr, _Nc>::size_type sw;
1413  for (sw = it; sw != mat.rows(); sw++) {
1414  if (mat.at(sw, it) != _Tp()) {
1415  break;
1416  }
1417  }
1418  if (sw != it) {
1419  mat.swap_row(sw, it);
1420  sign *= _Tp(-1);
1421  }
1422  for (typename estl::matrix::Matrix<_Tp, _Nr, _Nc>::size_type row = it + 1;
